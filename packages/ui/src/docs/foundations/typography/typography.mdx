export const SimpleTable = ({ headers, data }) => (
  <table>
    <thead>
      <tr>
        {headers.map((h) => (
          <th key={h}>{h}</th>
        ))}
      </tr>
    </thead>
    <tbody>
      {data.map((row, i) => (
        <tr key={i}>
          {row.map((cell, j) => (
            <td key={j}>{cell}</td>
          ))}
        </tr>
      ))}
    </tbody>
  </table>
);

export const TokenTable = ({ data }) => (
  <table>
    <thead>
      <tr>
        <th>トークン</th>
        <th>値</th>
      </tr>
    </thead>
    <tbody>
      {data.map(([token, value]) => (
        <tr key={token}>
          <td>
            <code>{token}</code>
          </td>
          <td>{value}</td>
        </tr>
      ))}
    </tbody>
  </table>
);

## 使用方法

### ユーティリティクラス

```tsx
<h1 className="dads-u-dsp-48B-140">大見出し</h1>
<p className="dads-u-std-16N-170">本文テキスト</p>
<code className="dads-u-mono-14N-150">console.log('Hello')</code>
```

### CSS変数

```css
.custom-heading {
  font-size: var(--font-size-24);
  font-weight: var(--font-weight-bold);
  line-height: var(--line-height-150);
  letter-spacing: var(--letter-spacing-2);
}
```
